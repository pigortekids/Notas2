CREATE TABLE CLIENTE (
	PRIMARY KEY ID INTEGER,
	ENDEREÇO VARCHAR(45),
	TELEFONE CHAR(9),
	E-MAIL VARCHAR(45)
);

CREATE TABLE PESSOA_FISICA (
	FOREIGN KEY IDCLIENTE REFERENCES CLIENTE(ID),
	PRIMARY KEY CPF CHAR(11)
);

CREATE TABLE PESSOA_JURIDICA (
	FOREIGN KEY IDCLIENTE REFERENCES CLIENTE(ID),
	PRIMARY KEY CNPJ CHAR(11)
);

CREATE TABLE COMPRA (
	FOREIGN KEY IDCLIENTE REFERENCES CLIENTE(ID),
	FOREIGN KEY IDLIVRO REFERENCES LIVRO(ID),
	DATA DATETIME
);

CREATE TABLE LIVRO (
	PRIMARY KEY ID INTEGER,
	AUTOR VARCHAR(45),
	ASSUNTO VARCHAR(45),
	QUANTIDADE INTEGER,
	ISBN CHAR(10),
	FOREIGN KEY IDEDITORA REFERENCES EDITORA(ID)
);

CREATE TABLE EDITORA (
	PRIMARY KEY ID INTEGER,
	ENDEREÇO VARCHAR(45),
	TELEFONE CHAR(9),
	GERENTE VARCHAR(45)
);
